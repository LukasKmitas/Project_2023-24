155void NeuralNetworks::backPropagation(const std::vector<float>& m_target, std::vector<std::vector<float>>& m_errors, const std::vector<std::vector<float>>& m_neuralNetwork, std::vector<std::vector<std::vector<float>>>& m_weights)
156{
157 for (int i = m_neuralNetwork.size() - 1; 0 < i; i--)
158 {
159 for (int j = 0; j < m_neuralNetwork[i].size(); j++)
160 {
161 int m_biasNeurons = 0;
162
163 if (i == m_neuralNetwork.size() - 1)
164 {
165 m_errors[i - 1][j] = pow(m_neuralNetwork[i][j] - m_target[j], 3);
166 }
167 else if (j >= BIAS_NEURONS[i])
168 {
169 int m_nextBiasNeuron = 0;
170
171 m_biasNeurons = BIAS_NEURONS[i];
172
173 if (i < m_neuralNetwork.size() - 2)
174 {
175 m_nextBiasNeuron = BIAS_NEURONS[1 + i];
176 }
177
178 m_errors[i - 1][j - m_biasNeurons] = 0;
179
180 for (int k = m_nextBiasNeuron; k < m_neuralNetwork[1 + i].size(); k++)
181 {
182 m_errors[i - 1][j - m_biasNeurons] += m_errors[i][k - m_nextBiasNeuron] * m_weights[i][k - m_nextBiasNeuron][j - m_biasNeurons];
183 }
184 }
185
186 if (j >= m_biasNeurons)
187 {
188 float m_neuronOutput = 0;
189
190 for (int k = 0; k < m_weights[i - 1][j - m_biasNeurons].size(); k++)
191 {
192 m_neuronOutput += m_neuralNetwork[i - 1][k] * m_weights[i - 1][j - m_biasNeurons][k];
193 }
194
195 for (int k = 0; k < m_neuralNetwork[i - 1].size(); k++)
196 {
197 m_weights[i - 1][j - m_biasNeurons][k] -= LEARNING_RATE * m_errors[i - 1][j - m_biasNeurons] * m_neuralNetwork[i - 1][k] * activationFunctionSigmoid(1, m_neuronOutput);
198 }
199 }
200 }

416float NeuralNetworks::activationFunctionSigmoid(bool m_derivative, float m_input)
417{
418 if (!m_derivative)
419 {
420 return 1.0f / (1.0f + std::exp(-m_input));
421 }
422 else
423 {
424 float sigmoid = 1.0f / (1.0f + std::exp(-m_input));
425 return sigmoid * (1 - sigmoid);
426 }
