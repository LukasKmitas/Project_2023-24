106void Unit::avoidCollisionsWithUnits(std::vector<Unit*>& m_allyUnits)
107{
108 const float minSeparation = 250.0f;
109 sf::Vector2f separationForce(0.0f, 0.0f);
110 int closeUnits = 0;
111
112 for (auto& unit : m_allyUnits)
113 {
114 if (unit == this) continue;
115
116 if ((m_unitTypeClass == UnitTypeClass::Air && (unit->m_unitTypeClass == UnitTypeClass::Vehicle || unit->m_unitTypeClass == UnitTypeClass::Infantry)) ||
117 (unit->m_unitTypeClass == UnitTypeClass::Air && (m_unitTypeClass == UnitTypeClass::Vehicle || m_unitTypeClass == UnitTypeClass::Infantry)))
118 {
119 continue;
120 }
121
122 float dist = distance(this->getPosition(), unit->getPosition());
123 if (dist < m_viewRadius - minSeparation)
124 {
125 sf::Vector2f pushAway = this->getPosition() - unit->getPosition();
126 pushAway = normalize(pushAway);
127 separationForce += pushAway / dist;
128 closeUnits++;
129 }
130 }
131
132 if (closeUnits > 0)
133 {
134 separationForce /= static_cast<float>(closeUnits);
135 separationForce = normalize(separationForce);
136 m_targetPosition += separationForce;
137 }

143void Unit::avoidCollisionsWithWalls()
144{
145 m_debugRays.clear();
146
147 if (!m_tiles || m_velocity == sf::Vector2f(0, 0))
148 {
149 return;
150 }
151
152 const float m_distanceAhead = 50.0f; // Distance to check ahead for walls
153 const float m_distanceSides = 20.0f; // Distance to check to the sides
154 sf::Vector2f m_currentPosition = getPosition();
155
156 std::vector<float> checkAngles = { 0, 30.0f, -30.0f, 60.0f, -60.0f, 90.0f, -90.0f };
157
158 for (float angle : checkAngles)
159 {
160 sf::Vector2f testDirection = rotateVector(normalize(m_velocity), angle);
161 float checkDistance = angle == 0 ? m_distanceAhead : m_distanceSides;
162 sf::Vector2f newPosition = m_currentPosition + testDirection * checkDistance;
163
164 m_debugRays.push_back(testDirection * checkDistance);
165
166 int newTileX = static_cast<int>(newPosition.x / (*m_tiles)[0][0].m_tileSize);
167 int newTileY = static_cast<int>(newPosition.y / (*m_tiles)[0][0].m_tileSize);
168
169 if (newTileX >= 0 && newTileX < (*m_tiles)[0].size() && newTileY >= 0 && newTileY < (*m_tiles).size() && (*m_tiles)[newTileY][newTileX].m_isWall)
170 {
171 sf::Vector2f avoidanceDirection = findAvoidanceDirection(m_currentPosition, m_distanceAhead);
172 if (avoidanceDirection != sf::Vector2f(0, 0))
173 {
174 m_velocity = lerp(m_velocity, normalize(avoidanceDirection) * m_speed, 0.1f); // Apply smoothing
175 break;
176 }

185void Unit::orientSpriteToMovement(sf::Time t_deltaTime)
186{
187 if (magnitude(m_velocity) > 0.0f)
188 {
189 float targetAngleRadians = std::atan2(m_velocity.y, m_velocity.x);
190 float targetAngleDegrees = toDegrees(targetAngleRadians) + 90;
191
192 float currentAngleDegrees = fmod(m_unitSprite.getRotation(), 360.0f);
193 if (currentAngleDegrees < 0)
194 {
195 currentAngleDegrees += 360.0f;
196 }
197 float difference = targetAngleDegrees - currentAngleDegrees;
198 if (difference > 180.0f)
199 {
200 difference -= 360.0f;
201 }
202 if (difference < -180.0f)
203 {
204 difference += 360.0f;
205 }
206
207 // Determine rotation speed and apply a fraction of the difference
208 float step = m_rotationSpeed * t_deltaTime.asSeconds();
209 if (std::abs(difference) < step)
210 {
211 step = std::abs(difference); // Clamp to avoid overshooting
212 }
213
214 m_unitSprite.setRotation(currentAngleDegrees + (difference > 0 ? step : -step));
215 }

487sf::Vector2f Unit::steerTowards(sf::Vector2f m_target)
488{
489 sf::Vector2f desiredVelocity = m_target - m_position;
490 float distance = magnitude(desiredVelocity);
491
492 if (distance > 0.1f)
493 {
494 desiredVelocity = normalize(desiredVelocity);
495 }
496
497 if (distance < m_slowingRadius)
498 {
499 desiredVelocity *= m_speed * (distance / m_slowingRadius);
500 }
501 else
502 {
503 desiredVelocity *= m_speed;
504 }
505
506 sf::Vector2f steer = desiredVelocity - m_velocity;
507
508 if (magnitude(steer) > m_maxForce)
509 {
510 steer = normalize(steer);
511 steer *= m_maxForce;
512 }
513
514 return steer;
