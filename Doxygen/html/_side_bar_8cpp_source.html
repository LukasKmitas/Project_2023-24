365void SideBar::drawGrid(sf::RenderWindow& m_window)
366{
367 float cellWidth = m_bottomBackground.getSize().x / m_gridCols;
368 float cellHeight = m_bottomBackground.getSize().y / m_gridRows;
369
370 sf::Color gridColor = sf::Color(63, 208, 212);
371 float gridThickness = 2.0f;
372
373 // Vertical lines
374 for (int i = 1; i < m_gridCols; ++i)
375 {
376 sf::RectangleShape line(sf::Vector2f(gridThickness, m_bottomBackground.getSize().y));
377 line.setPosition(m_bottomBackground.getPosition().x + i * cellWidth, m_bottomBackground.getPosition().y);
378 line.setFillColor(gridColor);
379 m_window.draw(line);
380 }
381
382 // Horizontal lines
383 for (int i = 1; i < m_gridRows; ++i)
384 {
385 sf::RectangleShape line(sf::Vector2f(m_bottomBackground.getSize().x, gridThickness));
386 line.setPosition(m_bottomBackground.getPosition().x, m_bottomBackground.getPosition().y + i * cellHeight);
387 line.setFillColor(gridColor);
388 m_window.draw(line);
389 }
390}
