158void TankAurora::movement(sf::Time t_deltaTime)
159{
160 float distance = magnitude(m_targetPosition - m_position);
161
162 if (distance < m_arrivalTolerance)
163 {
164 m_velocity = sf::Vector2f(0, 0);

167 else

174 speed *= (distance / m_slowingRadius);
175 }
176
177 sf::Vector2f desiredVelocity = desiredDirection * speed;
178 sf::Vector2f steeringForce = desiredVelocity - m_velocity;
179 steeringForce = normalize(steeringForce) * m_maxForce;
180
181 m_acceleration = steeringForce;
182 m_velocity += m_acceleration * t_deltaTime.asSeconds();
183
184 if (magnitude(m_velocity) > m_speed)

186 m_velocity = normalize(m_velocity) * m_speed;
