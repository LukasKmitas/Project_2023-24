64void Game::run()
65{
66 sf::Clock clock;

68 const float fps{ 60.0f };
69 sf::Time timePerFrame = sf::seconds(1.0f / fps);
70 while (m_window.isOpen())
71 {
72 processEvents();
73 timeSinceLastUpdate += clock.restart();

75 {
76 timeSinceLastUpdate -= timePerFrame;
77 processEvents(); // at least 60 fps

640void Game::updatePlayerAssets(sf::Time t_deltaTime)
641{
642 for (Building* playerBuilding : m_placedPlayerBuildings)
643 {
644 playerBuilding->update(t_deltaTime);
645 }
646
647 for (auto& unit : m_playerUnits)
648 {
649 unit->update(t_deltaTime, m_playerUnits);
650
651 for (auto& bullet : unit->m_bullets) // BULLETS
652 {
653 bool hit = false;
654
655 for (auto& enemyUnit : m_enemyUnits)
656 {
657 if (bullet.m_bulletShape.getGlobalBounds().intersects(enemyUnit->getSprite().getGlobalBounds()))
658 {
659 bullet.m_active = false;
660 enemyUnit->takeDamage(unit->getDamage());
661 if (enemyUnit->getHealth() <= 0)
662 {
663 enemyUnit->m_active = false;
664 }
665 spawnBulletSparkParticles(bullet.m_position);
666 }
667 }
668
669 for (auto& enemyBuilding : m_placedEnemyBuildings)
670 {
671 if (bullet.m_bulletShape.getGlobalBounds().intersects(enemyBuilding->getBuildingSprite().getGlobalBounds()))
672 {
673 bullet.m_active = false;
674 enemyBuilding->takeDamage(unit->getDamage());
675 hit = true;
676 spawnBulletSparkParticles(bullet.m_position);
677 break;
678 }
679 }
680 if (hit)
681 {
682 continue;
683 }
684 }
685 for (auto& missile : unit->m_missiles) // MISSILES
686 {
687 bool hit = false;
688
689 for (auto& enemyUnit : m_enemyUnits)
690 {
691 if (missile.m_active && missile.m_missileSprite.getGlobalBounds().intersects(enemyUnit->getSprite().getGlobalBounds()))
692 {
693 missile.m_active = false;
694 hit = true;
695 enemyUnit->takeDamage(missile.m_damage);
696 if (enemyUnit->getHealth() <= 0)
697 {
698 enemyUnit->m_active = false;
699 }
700 spawnExplosionParticle(missile.m_position);
701 SoundManager::getInstance().playSound("RocketExplosion");
702 break;
703 }
704 }
705 if (!hit)
706 {
707 for (auto& enemyBuilding : m_placedEnemyBuildings)
708 {
709 if (missile.m_active && missile.m_missileSprite.getGlobalBounds().intersects(enemyBuilding->getBuildingSprite().getGlobalBounds()))
710 {
711 missile.m_active = false;
712 enemyBuilding->takeDamage(unit->getDamage());
713 hit = true;
714 spawnExplosionParticle(missile.m_position);
715 SoundManager::getInstance().playSound("RocketExplosion");
716 break;
717 }
718 }
719 }
720 }
721
722 // Remove inactive bullets
723 unit->m_bullets.erase(std::remove_if(unit->m_bullets.begin(), unit->m_bullets.end(),
724 [](const Bullet& bullet) { return !bullet.m_active; }), unit->m_bullets.end());
725 // Remove inactive missiles
726 unit->m_missiles.erase(std::remove_if(unit->m_missiles.begin(), unit->m_missiles.end(),
727 [](const Missile& missile) { return !missile.m_active && missile.m_trail.empty(); }), unit->m_missiles.end());
728
729 // Remove dead enemy units
730 m_enemyUnits.erase(std::remove_if(m_enemyUnits.begin(), m_enemyUnits.end(),
731 [](Unit* enemyUnit) { return !enemyUnit->isActive(); }), m_enemyUnits.end());
732 // Remove buildings
733 m_placedEnemyBuildings.erase(std::remove_if(m_placedEnemyBuildings.begin(), m_placedEnemyBuildings.end(),
734 [](Building* building) { return building->getHealth() <= 0; }), m_placedEnemyBuildings.end());
735 }

787 m_random.seed(std::chrono::system_clock::now().time_since_epoch().count());

834void Game::updateNeuralNetwork()
835{
836 if (m_train) // Training the neural network
837 {
838 m_totalErrors = 0;
839
840 for (int i = 0; i < TRAININGS_PER_FRAME; i++)
841 {
842 int m_inputIndex = rand() % m_neural_network.getInputs().size();
843
844 m_neural_network.forwardPropagation(true, m_neural_network.getInputs()[m_inputIndex], m_neuralNetwork, m_weights);
845 m_neural_network.backPropagation(m_neural_network.getTargetOutputs()[m_inputIndex], m_errors, m_neuralNetwork, m_weights);
846 }
847
848 for (int i = 0; i < m_neural_network.getInputs().size(); i++)
849 {
850 vector_1d m_outputs = m_neural_network.forwardPropagation(false, m_neural_network.getInputs()[i], m_neuralNetwork, m_weights);
851
852 for (int j = 0; j < m_outputs.size(); j++)
853 {
854 m_totalErrors += std::abs(m_outputs[j] - m_neural_network.getTargetOutputs()[i][j]);
855 }
856 }
857 }
858 m_outputTexture.update(m_outputImage);
859}
